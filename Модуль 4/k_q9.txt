//Из регистра «ИсторияЗаказов» вывести 
//- товар,
//- количество товара,
//- среднюю цену на товар, рассчитанную как среднее арифметическое,
//- среднюю цену, рассчитанную по формуле сумма деленная на количество товара
//- сумма по всем заказам за товар
//за весь период учета хозяйственной деятельности.

ВЫБРАТЬ //подготовка данных
	ИсторияЗаказов.Товар.Наименование КАК Товар,
	SUM(ИсторияЗаказов.Количество ) КАК Количество,
	COUNT(ИсторияЗаказов.Товар.Наименование) AS КоличествоСтрок,
	AVG(ИсторияЗаказов.Цена) КАК Цена,
	SUM(ИсторияЗаказов.Сумма) КАК Сумма 
INTO T1 ИЗ
	РегистрНакопления.ИсторияЗаказов КАК ИсторияЗаказов
GROUP BY
	ИсторияЗаказов.Товар.Наименование;
//вывод результата
SELECT 
	Товар, Количество,
	Цена, 
	Сумма/Количество AS ЦенаРасчётная, 
	Сумма	
FROM 
	T1
