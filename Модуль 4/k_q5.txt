//5. Вывести названия стран-производителей товаров и для каждой страны – количество товаров,
//имеющихся в продаже из этой страны. Упорядочить по убыванию количества товара
SELECT
	Товары.Страна.Наименование AS Страна,		
	Count(Товары.Наименование) AS КоличествоРазличныхТоваров
FROM
	Справочник.Товары КАК Товары
GROUP BY
	Товары.Страна.Наименование
ORDER BY
	Count(Товары.Наименование) DESC
