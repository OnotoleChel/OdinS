//5. Вывести названия Покупателей, которые работают в городах Москва и Санкт-Петербург
//Вывести в одном столбце название Покупателя, в другом – букву «М», если покупатель работает в г. Москва
//или «СПБ», если покупатель работает в г. Санкт-Петербург.

SELECT 
	Покупатели.Наименование AS Наименование,
	//Покупатели.Регион.Наименование AS РегионНаименование
	CASE
		WHEN Покупатели.Регион.Наименование = "г. Москва" THEN
			"M"
		WHEN Покупатели.Регион.Наименование = "г. Санкт-Петербург" THEN
			"СПБ"
		END
	AS РегионНаименование

FROM 
	Справочник.Покупатели AS Покупатели
WHERE
	Покупатели.Регион.Наименование IN ("г. Москва", "г. Санкт-Петербург")
