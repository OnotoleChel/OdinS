//2. Для всех договоров покупателей, заключенных в 2021 году, вывести номер договора, дата
//заключения договора, наименование договора, наименование покупателей, указанных в договоре.
//Результат выборки упорядочить по дате договора.

SELECT
	Номер, 
	Дата, 
	Наименование AS НаименованиеДоговора, 
	Покупатели.Наименование AS НаименованиеПокупателя 
FROM
	Справочник.ДоговорыПокупателей КАК ДоговорыПокупателей
WHERE
	YEAR(Дата) = &MyYear
ORDER BY
	Дата
