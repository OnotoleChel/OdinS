//Вывести по всем совершенным заказам за 2021 год: название товаров, количество заказанных
//товаров, среднюю цену за товар и сумму, оплаченную за все заказы соответствующего товара.

ВЫБРАТЬ
	Товар.Наименование,
	sum(Количество),
	avg(Цена) AS СредняяЦена,
	sum(Сумма)
ИЗ
	РегистрНакопления.ИсторияЗаказов AS ИсторияЗаказов
WHERE
	YEAR(Период) = &MyYear
GROUP BY 
 	Товар.Наименование
