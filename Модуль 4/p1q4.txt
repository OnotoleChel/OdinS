//4. Вывести названия Покупателей, у которых на звание начинается на букву С
//помимо названия организации в выборку добавить название региона и вывести в
//формате «Регион: <Название региона>» и «Покупатель: <Наименование покупателя>»
//где <Название региона> и <Наименование покупателя> - это соответствующие данные, полученные из запроса 
SELECT 
	"Покупатель: " + Наименование AS Покупатель
	, "Регион: " + Регион.Наименование AS Регион
FROM
	Справочник.Покупатели
WHERE
	Справочник.Покупатели.Наименование LIKE &MyPattern
